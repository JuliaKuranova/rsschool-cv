##Code example
###Task from [Leetcode](https://leetcode.com/problems/count-odd-numbers-in-an-interval-range/)
```javascript
/**
 * @param {number} low
 * @param {number} high
 * @return {number}
 */
var countOdds = function(low, high) {
    let countOfAdd = 0;
    for (let i = low; i <= high; i++) {
        if (i % 2 !== 0) {
           countOfAdd++
        }
    }
    return countOfAdd
};
```
